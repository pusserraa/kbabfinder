    <script src="https://unpkg.com/react@18/umd/react.production.min.js"></script>

    <script src="https://unpkg.com/@babel/standalone/babel.min.js"></script>


    <script type="text/babel">
        const { useState, useEffect, useRef } = React;


        const apiKey = "";


        const Icon = ({ name, className = "w-5 h-5" }) => {
            const icons = {
                menu: <path d="M4 6h16M4 12h16M4 18h16" />,
                search: <React.Fragment><circle cx="11" cy="11" r="8" /><path d="m21 21-4.3-4.3" /></React.Fragment>,
                x: <path d="M18 6 6 18M6 6l12 12" />,
                star: <path d="M12 2l3.09 6.26L22 9.27l-5 4.87 1.18 6.88L12 17.77l-6.18 3.25L7 14.14 2 9.27l6.91-1.01L12 2z" />,
                navigation: <path d="M3 11l19-9-9 19-2-8-8-2z" />,
                arrowLeft: <path d="m12 19-7-7 7-7M5 12h14" />,
                bookPlus: <React.Fragment><path d="M4 19.5v-15A2.5 2.5 0 0 1 6.5 2H20v20H6.5a2.5 2.5 0 0 1 0-5H20" /><path d="M12 7v6" /><path d="M9 10h6" /></React.Fragment>,
                bookCheck: <React.Fragment><path d="M4 19.5v-15A2.5 2.5 0 0 1 6.5 2H20v20H6.5a2.5 2.5 0 0 1 0-5H20" /><path d="m9 10 2 2 4-4" /></React.Fragment>,
                user: <React.Fragment><path d="M19 21v-2a4 4 0 0 0-4-4H9a4 4 0 0 0-4 4v2" /><circle cx="12" cy="7" r="4" /></React.Fragment>,
                chevronRight: <path d="m9 18 6-6-6-6" />,
                settings: <React.Fragment><path d="M12.22 2h-.44a2 2 0 0 0-2 2a2 2 0 0 1-2 2a2 2 0 0 0-2 2v.44a2 2 0 0 0 2 2a2 2 0 0 1 2 2a2 2 0 0 0 2 2h.44a2 2 0 0 0 2-2a2 2 0 0 1 2-2a2 2 0 0 0 2-2V7.78a2 2 0 0 0-2-2a2 2 0 0 1-2-2a2 2 0 0 0-2-2ZM12 15a3 3 0 1 0 0-6a3 3 0 0 0 0 6Z"/></React.Fragment>,
                check: <path d="M20 6L9 17l-5-5" />,
                camera: <React.Fragment><path d="M14.5 4h-5L7 7H4a2 2 0 0 0-2 2v9a2 2 0 0 0 2 2h16a2 2 0 0 0 2-2V9a2 2 0 0 0-2-2h-3l-2.5-3z"/><circle cx="12" cy="13" r="3"/></React.Fragment>,
                upload: <path d="M21 15v4a2 2 0 0 1-2 2H5a2 2 0 0 1-2-2v-4M17 8l-5-5-5 5M12 3v12" />
            };


            return (
                <svg viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="2" strokeLinecap="round" strokeLinejoin="round" className={className}>
                    {icons[name]}
                </svg>
            );
        };


        const translations = {
            ca: {
                search: "Busca el millor durum",
                placeholder: "Ciutat (ex: Barcelona...)",
                back: "Tornar",
                reviews: "Ressenyes",
                write_review: "Escriu la teva opinió",
                publish: "Publicar",
                arrival: "Com arribar",
                profile: "Perfil Kebaber",
                language: "Idioma",
                scanning: "Escanejant...",
                view_details: "Detalls +",
                rating: "Puntuació:",
                my_album: "El meu Àlbum",
                empty_album: "Encara no has afegit cap kebab!",
                added: "Afegit a l'àlbum",
                view_all: "Veure tota la col·lecció",
                settings: "Configuració",
                edit_profile: "Editar Perfil",
                name_label: "Nom",
                nickname_label: "Apodo / Mote",
                avatar_label: "Foto de perfil",
                save_changes: "Guardar canvis",
                upload_photo: "Pujar foto"
            }
        };


        const AVATARS = [
            "https://api.dicebear.com/7.x/avataaars/svg?seed=Felix",
            "https://api.dicebear.com/7.x/bottts/svg?seed=Kebab1",
            "https://api.dicebear.com/7.x/fun-emoji/svg?seed=Durum",
            "https://api.dicebear.com/7.x/shapes/svg?seed=Spicy"
        ];


        const App = () => {
            const [view, setView] = useState('home');
            const [lang, setLang] = useState('ca');
            const [drawerOpen, setDrawerOpen] = useState(false);
            const [isEditingProfile, setIsEditingProfile] = useState(false);
            const [userData, setUserData] = useState({
                name: "Kebab Master",
                nickname: "Durum Mixt",
                avatar: AVATARS[0],
                collection: []
            });
            const [editForm, setEditForm] = useState({ ...userData });
            const [selectedKebab, setSelectedKebab] = useState(null);
            const [query, setQuery] = useState('');
            const [loading, setLoading] = useState(false);
            const [results, setResults] = useState([]);
            const [reviews, setReviews] = useState({});
            const [newReview, setNewReview] = useState({ rating: 5, text: '' });
            const [aiInsight, setAiInsight] = useState('');
            const [aiLoading, setAiLoading] = useState(false);
           
            const fileInputRef = useRef(null);


            const t = translations[lang] || translations['ca'];


            const toggleCollection = (kebab) => {
                const exists = userData.collection.find(item => (item.id || item.nom) === (kebab.id || kebab.nom));
                if (exists) {
                    setUserData(prev => ({ ...prev, collection: prev.collection.filter(item => (item.id || item.nom) !== (kebab.id || kebab.nom)) }));
                } else {
                    setUserData(prev => ({ ...prev, collection: [...prev.collection, kebab] }));
                }
            };


            const isKebabInCollection = (kebab) => {
                if (!kebab) return false;
                return userData.collection.some(item => (item.id || item.nom) === (kebab.id || kebab.nom));
            };


            const handleSaveProfile = (e) => {
                e.preventDefault();
                setUserData({ ...userData, ...editForm });
                setIsEditingProfile(false);
            };


            const handleFileUpload = (e) => {
                const file = e.target.files[0];
                if (file) {
                    const reader = new FileReader();
                    reader.onloadend = () => {
                        setEditForm({ ...editForm, avatar: reader.result });
                    };
                    reader.readAsDataURL(file);
                }
            };


            const fetchWithRetry = async (url, options, retries = 5, backoff = 1000) => {
                try {
                    const res = await fetch(url, options);
                    if (!res.ok) throw new Error('Status ' + res.status);
                    return await res.json();
                } catch (err) {
                    if (retries > 0) {
                        await new Promise(r => setTimeout(r, backoff));
                        return fetchWithRetry(url, options, retries - 1, backoff * 2);
                    }
                    throw err;
                }
            };


            const searchKebabs = async (location) => {
                if (!location) return;
                setLoading(true);
                const userQuery = `Search 6 kebab restaurants in ${location}. Respond ONLY with pure JSON: {"kebabs": [{"id": "${Math.random()}", "nom": "Name", "valoracio": 4.5, "adreca": "Street X", "descripcio": "Good meat"}]}`;
                try {
                    const data = await fetchWithRetry(`https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${apiKey}`, {
                        method: 'POST',
                        headers: { 'Content-Type': 'application/json' },
                        body: JSON.stringify({ contents: [{ parts: [{ text: userQuery }] }], tools: [{ "google_search": {} }] })
                    });
                    const text = data.candidates?.[0]?.content?.parts?.[0]?.text || "{}";
                    const clean = text.replace(/```json/g, '').replace(/```/g, '').trim();
                    const parsed = JSON.parse(clean);
                    setResults(parsed.kebabs || []);
                } catch (err) { console.error(err); }
                finally { setLoading(false); }
            };


            const getAiInsight = async (kebab) => {
                setAiLoading(true);
                try {
                    const data = await fetchWithRetry(`https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${apiKey}`, {
                        method: 'POST',
                        headers: { 'Content-Type': 'application/json' },
                        body: JSON.stringify({ contents: [{ parts: [{ text: `One sentence summary of ${kebab.nom} in ${lang}.` }] }] })
                    });
                    setAiInsight(data.candidates[0].content.parts[0].text);
                } catch (e) { setAiInsight("Popular choice."); }
                finally { setAiLoading(false); }
            };


            const addReview = (e) => {
                e.preventDefault();
                if (!newReview.text) return;
                const id = selectedKebab.id || selectedKebab.nom;
                const review = {
                    user: userData.name,
                    text: newReview.text,
                    rating: newReview.rating,
                    date: new Date().toLocaleDateString()
                };
                setReviews(prev => ({ ...prev, [id]: [review, ...(prev[id] || [])] }));
                setNewReview({ rating: 5, text: '' });
            };


            return (
                <div className="min-h-screen">
                    {/* Drawer/Sidebar */}
                    {drawerOpen && (
                        <div className="fixed inset-0 z-50 flex">
                            <div className="drawer-overlay absolute inset-0" onClick={() => {setDrawerOpen(false); setIsEditingProfile(false);}}></div>
                            <div className="drawer-content relative w-80 bg-white h-full shadow-2xl p-6 flex flex-col overflow-y-auto animate-fade-in">
                               
                                {isEditingProfile ? (
                                    <div className="animate-fade-in">
                                        <div className="flex justify-between items-center mb-8">
                                            <h2 className="text-xl font-black uppercase tracking-tighter">{t.edit_profile}</h2>
                                            <button onClick={() => setIsEditingProfile(false)} className="p-2 bg-gray-50 rounded-full"><Icon name="x" /></button>
                                        </div>
                                        <form onSubmit={handleSaveProfile} className="space-y-6">
                                            <div>
                                                <label className="text-[10px] font-black uppercase text-gray-400 block mb-2 tracking-widest">{t.avatar_label}</label>
                                               
                                                <div className="flex flex-col items-center gap-4 mb-4">
                                                    <div className="w-24 h-24 bg-gray-100 rounded-3xl overflow-hidden border-4 border-orange-50 shadow-inner">
                                                        <img src={editForm.avatar} className="w-full h-full object-cover" />
                                                    </div>
                                                    <input
                                                        type="file"
                                                        ref={fileInputRef}
                                                        className="hidden"
                                                        accept="image/*"
                                                        onChange={handleFileUpload}
                                                    />
                                                    <button
                                                        type="button"
                                                        onClick={() => fileInputRef.current.click()}
                                                        className="flex items-center gap-2 px-4 py-2 bg-orange-50 text-orange-600 rounded-xl font-black text-[10px] uppercase tracking-widest border border-orange-100"
                                                    >
                                                        <Icon name="upload" className="w-3 h-3" />
                                                        {t.upload_photo}
                                                    </button>
                                                </div>


                                                <p className="text-[9px] font-bold text-gray-300 uppercase tracking-widest text-center mb-2">O tria un d'aquests</p>
                                                <div className="grid grid-cols-4 gap-2">
                                                    {AVATARS.map((av, idx) => (
                                                        <button
                                                            type="button"
                                                            key={idx}
                                                            onClick={() => setEditForm({...editForm, avatar: av})}
                                                            className={`p-1 rounded-xl border-2 transition-all ${editForm.avatar === av ? 'border-orange-600 scale-110' : 'border-transparent'}`}
                                                        >
                                                            <img src={av} alt="Avatar" className="w-full aspect-square rounded-lg bg-gray-100" />
                                                        </button>
                                                    ))}
                                                </div>
                                            </div>
                                            <div>
                                                <label className="text-[10px] font-black uppercase text-gray-400 block mb-2 tracking-widest">{t.name_label}</label>
                                                <input
                                                    type="text"
                                                    value={editForm.name}
                                                    onChange={e => setEditForm({...editForm, name: e.target.value})}
                                                    className="w-full bg-gray-50 p-4 rounded-xl border-2 border-transparent focus:border-orange-500 outline-none font-bold"
                                                />
                                            </div>
                                            <div>
                                                <label className="text-[10px] font-black uppercase text-gray-400 block mb-2 tracking-widest">{t.nickname_label}</label>
                                                <input
                                                    type="text"
                                                    value={editForm.nickname}
                                                    onChange={e => setEditForm({...editForm, nickname: e.target.value})}
                                                    className="w-full bg-gray-50 p-4 rounded-xl border-2 border-transparent focus:border-orange-500 outline-none font-bold"
                                                />
                                            </div>
                                            <button type="submit" className="w-full bg-orange-600 text-white py-4 rounded-2xl font-black text-xs uppercase tracking-widest shadow-lg shadow-orange-100">{t.save_changes}</button>
                                        </form>
                                    </div>
                                ) : (
                                    <React.Fragment>
                                        <div className="flex justify-between items-center mb-8">
                                            <h2 className="text-xl font-black uppercase tracking-tighter">{t.profile}</h2>
                                            <button onClick={() => setDrawerOpen(false)} className="p-2 bg-gray-50 rounded-full"><Icon name="x" /></button>
                                        </div>
                                        <div className="space-y-8">
                                            <div className="relative group">
                                                <div className="flex flex-col items-center gap-3 p-6 bg-orange-600 rounded-[2.5rem] shadow-xl text-white">
                                                    <div className="w-20 h-20 bg-white rounded-3xl flex items-center justify-center overflow-hidden shadow-inner border-4 border-white/20">
                                                        <img src={userData.avatar} alt="Profile" className="w-full h-full object-cover" />
                                                    </div>
                                                    <div className="text-center">
                                                        <p className="font-black text-xl">{userData.name}</p>
                                                        <p className="text-[10px] uppercase text-orange-200 font-black tracking-widest">{userData.nickname}</p>
                                                    </div>
                                                </div>
                                                <button
                                                    onClick={() => {setEditForm({...userData}); setIsEditingProfile(true);}}
                                                    className="absolute -top-2 -right-2 p-3 bg-white text-orange-600 rounded-2xl shadow-xl hover:scale-110 active:scale-95 transition-all border border-orange-50"
                                                >
                                                    <Icon name="settings" className="w-5 h-5" />
                                                </button>
                                            </div>


                                            <button
                                                onClick={() => { setView('album'); setDrawerOpen(false); }}
                                                className="w-full flex items-center justify-between p-4 bg-orange-50 rounded-2xl hover:bg-orange-100 transition-colors group"
                                            >
                                                <div className="flex items-center gap-3">
                                                    <div className="p-2 bg-orange-600 rounded-lg text-white">
                                                        <Icon name="bookPlus" className="w-5 h-5" />
                                                    </div>
                                                    <span className="font-black text-sm uppercase">{t.my_album}</span>
                                                </div>
                                                <div className="flex items-center gap-2">
                                                    <span className="bg-orange-200 text-orange-700 px-2 py-0.5 rounded-full text-[10px] font-black">{userData.collection.length}</span>
                                                    <Icon name="chevronRight" className="w-4 h-4 text-orange-400 group-hover:translate-x-1 transition-transform" />
                                                </div>
                                            </button>


                                            <div className="pt-4 border-t border-gray-100">
                                                <p className="text-[10px] font-black uppercase text-gray-400 mb-4 tracking-widest">{t.language}</p>
                                                <div className="grid grid-cols-2 gap-2">
                                                    {['ca', 'es', 'en', 'fr'].map(l => (
                                                        <button key={l} onClick={() => setLang(l)} className={`py-3 rounded-xl text-[10px] font-black uppercase tracking-widest transition-all ${lang === l ? 'bg-black text-white scale-105 shadow-lg' : 'bg-gray-100 hover:bg-gray-200 text-gray-400'}`}>{l}</button>
                                                    ))}
                                                </div>
                                            </div>
                                        </div>
                                    </React.Fragment>
                                )}
                            </div>
                        </div>
                    )}


                    <header className="bg-white/80 backdrop-blur-md border-b sticky top-0 z-20 p-4 flex items-center justify-between">
                        <button onClick={() => setDrawerOpen(true)} className="p-2 text-orange-600 hover:bg-orange-50 rounded-full transition-colors flex items-center gap-3">
                            <img src={userData.avatar} className="w-8 h-8 rounded-lg bg-orange-100 object-cover" alt="Avatar" />
                            <Icon name="menu" />
                        </button>
                        <div className="flex items-center gap-2 cursor-pointer" onClick={() => setView('home')}>
                            <div className="bg-orange-600 p-1.5 rounded-lg text-white shadow-lg"><Icon name="star" className="w-4 h-4" /></div>
                            <h1 className="text-xl font-black tracking-tighter uppercase">KebabFinder</h1>
                        </div>
                        <div className="w-10"></div>
                    </header>


                    {view === 'home' && (
                        <main className="max-w-4xl mx-auto px-4 py-10 animate-fade-in">
                            <div className="text-center mb-12">
                                <h2 className="text-5xl font-black mb-6 tracking-tighter leading-none">{t.search}</h2>
                                <form onSubmit={(e) => { e.preventDefault(); searchKebabs(query); }} className="relative max-w-md mx-auto group">
                                    <input type="text" value={query} onChange={e => setQuery(e.target.value)} placeholder={t.placeholder} className="w-full px-6 py-5 rounded-full border-2 border-orange-100 outline-none focus:border-orange-500 shadow-xl transition-all group-hover:shadow-orange-200/50" />
                                    <button className="absolute right-3 top-3 bottom-3 bg-orange-600 text-white px-6 rounded-full hover:bg-orange-700 transition-colors shadow-lg active:scale-95"><Icon name="search" /></button>
                                </form>
                            </div>


                            {loading && <div className="flex flex-col items-center py-10"><div className="kebab-loader mb-4"></div><p className="text-[10px] font-black uppercase tracking-widest text-orange-600">{t.scanning}</p></div>}


                            <div className="grid grid-cols-1 md:grid-cols-2 gap-6">
                                {results.map((k, i) => (
                                    <div key={i} onClick={() => { setSelectedKebab(k); setView('detail'); getAiInsight(k); }} className="bg-white p-6 rounded-[2.5rem] border border-gray-100 shadow-sm hover:shadow-2xl hover:border-orange-200 transition-all cursor-pointer group animate-fade-in" style={{animationDelay: `${i * 0.1}s`}}>
                                        <div className="flex justify-between items-start mb-2">
                                            <h3 className="font-black text-xl tracking-tight leading-tight group-hover:text-orange-600 transition-colors">{k.nom}</h3>
                                            {isKebabInCollection(k) && <div className="p-1.5 bg-orange-100 rounded-lg text-orange-600"><Icon name="bookCheck" className="w-5 h-5" /></div>}
                                        </div>
                                        <p className="text-[10px] text-gray-400 font-bold uppercase mb-4 tracking-wider">{k.adreca}</p>
                                        <div className="flex justify-between items-center">
                                            <span className="bg-yellow-50 text-yellow-600 px-3 py-1 rounded-full font-black text-xs flex items-center gap-1 border border-yellow-100">⭐ {k.valoracio}</span>
                                            <span className="text-orange-600 text-[10px] font-black uppercase group-hover:translate-x-1 transition-transform">{t.view_details}</span>
                                        </div>
                                    </div>
                                ))}
                            </div>
                        </main>
                    )}


                    {view === 'album' && (
                        <main className="max-w-4xl mx-auto px-4 py-8 animate-fade-in">
                            <button onClick={() => setView('home')} className="flex items-center gap-2 text-gray-400 hover:text-orange-600 font-black text-sm mb-6 transition-colors"><Icon name="arrowLeft" /> {t.back}</button>
                            <h2 className="text-4xl font-black mb-8 tracking-tighter uppercase">{t.my_album}</h2>
                           
                            {userData.collection.length === 0 ? (
                                <div className="text-center py-20 bg-white rounded-[3rem] border-2 border-dashed border-orange-100">
                                    <div className="bg-orange-50 w-20 h-20 rounded-full flex items-center justify-center mx-auto mb-4">
                                        <Icon name="bookPlus" className="text-orange-300 w-10 h-10" />
                                    </div>
                                    <p className="text-gray-400 font-bold uppercase text-xs tracking-widest">{t.empty_album}</p>
                                </div>
                            ) : (
                                <div className="album-grid">
                                    {userData.collection.map((k, i) => (
                                        <div key={i} onClick={() => { setSelectedKebab(k); setView('detail'); getAiInsight(k); }} className="bg-white p-6 rounded-[2.5rem] border border-gray-100 shadow-md hover:shadow-xl transition-all cursor-pointer group border-l-4 border-l-orange-600">
                                            <div className="flex justify-between items-start mb-2">
                                                <h3 className="font-black text-lg truncate group-hover:text-orange-600">{k.nom}</h3>
                                                <Icon name="bookCheck" className="text-orange-600 w-4 h-4" />
                                            </div>
                                            <p className="text-[9px] text-gray-400 font-bold uppercase truncate mb-4">{k.adreca}</p>
                                            <div className="flex items-center gap-2">
                                                <span className="text-yellow-500 font-black text-xs">⭐ {k.valoracio}</span>
                                                <div className="h-1 w-1 bg-gray-200 rounded-full"></div>
                                                <span className="text-orange-600 text-[10px] font-black uppercase">{t.view_details}</span>
                                            </div>
                                        </div>
                                    ))}
                                </div>
                            )}
                        </main>
                    )}


                    {view === 'detail' && (
                        <div className="max-w-2xl mx-auto px-4 py-8 animate-fade-in">
                            <div className="flex justify-between items-center mb-6">
                                <button onClick={() => setView('home')} className="flex items-center gap-2 text-gray-400 hover:text-black font-bold text-sm transition-colors"><Icon name="arrowLeft" /> {t.back}</button>
                                <button
                                    onClick={() => toggleCollection(selectedKebab)}
                                    className={`p-4 rounded-3xl transition-all shadow-lg active:scale-90 ${isKebabInCollection(selectedKebab) ? 'bg-orange-600 text-white scale-105' : 'bg-white text-orange-600 border border-orange-100 hover:bg-orange-50'}`}
                                >
                                    <Icon name={isKebabInCollection(selectedKebab) ? "bookCheck" : "bookPlus"} className="w-6 h-6" />
                                </button>
                            </div>


                            <div className="bg-white rounded-[3rem] shadow-2xl overflow-hidden mb-8 border border-gray-100">
                                <div className="h-48 bg-orange-600 flex flex-col items-center justify-center relative overflow-hidden">
                                    <div className="absolute inset-0 opacity-20 bg-[radial-gradient(circle,white_1px,transparent_1px)] bg-[size:30px_30px]"></div>
                                    <div className="absolute top-0 left-0 w-full h-full bg-gradient-to-b from-black/20 to-transparent"></div>
                                    <h2 className="text-4xl font-black text-white px-6 text-center z-10 tracking-tighter leading-none">{selectedKebab.nom}</h2>
                                    {isKebabInCollection(selectedKebab) && <p className="text-[10px] font-black uppercase text-orange-200 mt-3 z-10 tracking-[0.2em] bg-white/10 backdrop-blur-md px-4 py-1 rounded-full">{t.added}</p>}
                                </div>
                                <div className="p-8">
                                    <div className="bg-orange-50 p-6 rounded-[2rem] mb-8 border border-orange-100 relative shadow-inner">
                                        <div className="absolute -top-3 left-8 bg-orange-600 text-white px-3 py-1 rounded-full text-[9px] font-black uppercase tracking-widest shadow-lg">AI INSIGHT</div>
                                        <p className="text-base font-bold italic text-gray-700 leading-relaxed pt-2">"{aiLoading ? '...' : aiInsight}"</p>
                                    </div>
                                    <button onClick={() => window.open(`https://maps.google.com/?q=${selectedKebab.nom} ${selectedKebab.adreca}`)} className="w-full bg-black text-white py-5 rounded-2xl font-black text-xs uppercase tracking-[0.2em] flex items-center justify-center gap-3 hover:bg-gray-800 transition-all shadow-xl active:translate-y-1"><Icon name="navigation" /> {t.arrival}</button>
                                </div>
                            </div>


                            <div className="bg-white rounded-[3rem] p-8 shadow-sm border border-gray-100">
                                <h3 className="text-2xl font-black mb-6 tracking-tight uppercase">{t.reviews}</h3>
                                <form onSubmit={addReview} className="mb-10 space-y-6">
                                    <div className="flex items-center justify-between bg-gray-50 p-4 rounded-2xl">
                                        <span className="text-[10px] font-black uppercase text-gray-400 tracking-widest">{t.rating}</span>
                                        <div className="star-rating">
                                            {[5, 4, 3, 2, 1].map(num => (
                                                <React.Fragment key={num}>
                                                    <input type="radio" id={`star-${num}`} name="rating" value={num} checked={newReview.rating === num} onChange={() => setNewReview({...newReview, rating: num})} />
                                                    <label htmlFor={`star-${num}`} className="text-2xl">★</label>
                                                </React.Fragment>
                                            ))}
                                        </div>
                                    </div>
                                    <textarea
                                        value={newReview.text}
                                        onChange={e => setNewReview({...newReview, text: e.target.value})}
                                        className="w-full bg-gray-50 rounded-[1.5rem] p-5 text-sm outline-none h-24 border-2 border-transparent focus:border-orange-200 focus:bg-white transition-all resize-none shadow-inner"
                                        placeholder={t.write_review}
                                    ></textarea>
                                    <button className="bg-orange-600 text-white px-10 py-4 rounded-2xl font-black text-[10px] uppercase tracking-widest shadow-lg shadow-orange-200 hover:bg-orange-700 active:scale-95 transition-all">{t.publish}</button>
                                </form>
                                <div className="space-y-8">
                                    {(reviews[selectedKebab.id || selectedKebab.nom] || []).map((r, i) => (
                                        <div key={i} className="border-b border-gray-50 pb-6 last:border-0 animate-fade-in">
                                            <div className="flex justify-between items-center mb-3">
                                                <div className="flex items-center gap-3">
                                                    <div className="w-8 h-8 bg-orange-100 text-orange-600 rounded-xl flex items-center justify-center text-xs font-black shadow-sm">
                                                        {r.user[0]}
                                                    </div>
                                                    <p className="font-bold text-sm tracking-tight">{r.user}</p>
                                                </div>
                                                <div className="flex text-yellow-500 text-xs">
                                                    {[...Array(r.rating)].map((_, i) => <span key={i}>★</span>)}
                                                </div>
                                            </div>
                                            <p className="text-sm text-gray-600 ml-11 leading-relaxed">{r.text}</p>
                                            <p className="text-[9px] text-gray-400 mt-3 font-bold uppercase tracking-widest ml-11">{r.date}</p>
                                        </div>
                                    ))}
                                </div>
                            </div>
                        </div>
                    )}
                </div>
            );
        };


        const root = ReactDOM.createRoot(document.getElementById('app'));
        root.render(<App />);
